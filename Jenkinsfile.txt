pipeline {
    agent any
    options {
        timeout(time: 10, unit: 'MINUTES')
    }
    stages {
        stage('install') {
            steps {
                bat 'npm install'

            }
        }
        stage('build') {
                    steps {

                        bat 'npm run-script build'
                    }
                }
        stage('tests without running app - independent') {
            parallel {
                        stage('test-unit') {
                            steps {
                                bat 'npm run-script test'
                            }
                        }
                        stage('test-component') {
                            steps {
                                bat 'npm run-script test-jasmine'
                            }
                        }
                         stage('test-integration') {
                             steps {
                                 bat 'npm run-script test-old'
                             }
                         }
                        stage('start-app') {
                            steps {
                                bat 'npm start'
                            }
                        }
                        stage('test-GUI') {
                            steps {
                                bat 'npm run-script test-selenium'
                            }
                        }
            }
        }


    }
}
